<script>
const canvas = document.querySelector("#space");
const ctx = canvas.getContext("2d");

const controls = {
collectors: document.querySelector("#collectors"),
orbit: document.querySelector("#orbit"),
speed: document.querySelector("#speed"),
efficiency: document.querySelector("#efficiency")
};

const labels = {
collectors: document.querySelector("#collectorValue"),
orbit: document.querySelector("#orbitValue"),
speed: document.querySelector("#speedValue"),
efficiency: document.querySelector("#efficiencyValue"),
energy: document.querySelector("#energyText"),
power: document.querySelector("#powerText")
};

let width = 0;
let height = 0;
let time = 0;
let stars = [];

function resize() {
const ratio = window.devicePixelRatio || 1;
width = canvas.clientWidth;
height = canvas.clientHeight;
canvas.width = Math.floor(width * ratio);
canvas.height = Math.floor(height * ratio);
ctx.setTransform(ratio, 0, 0, ratio, 0, 0);
stars = makeStars(130);
}

function makeStars(count) {
return Array.from({ length: count }, () => ({
x: Math.random() * width,
y: Math.random() * height,
r: Math.random() * 1.4 + 0.2,
glow: Math.random() * 0.6 + 0.25
}));
}

function readSettings() {
return {
collectors: Number(controls.collectors.value),
orbit: Number(controls.orbit.value),
speed: Number(controls.speed.value),
efficiency: Number(controls.efficiency.value)
};
}

function updateLabels(settings) {
const keplerSpeed = settings.speed * Math.sqrt(180 / settings.orbit);

labels.collectors.value = settings.collectors;
labels.orbit.value = settings.orbit;
labels.speed.value = keplerSpeed.toFixed(2) + "x";
labels.efficiency.value = settings.efficiency + "%";

const coverage = 1 - Math.exp(-settings.collectors / 190);
const distanceLoss = 180 / settings.orbit;
const captured = coverage * (settings.efficiency / 100) * distanceLoss;
const percent = Math.min(captured * 100, 99.9);
const terawatts = percent * 38.46;

labels.energy.textContent = percent.toFixed(1) + "%";
labels.power.textContent = terawatts.toFixed(2) + " TW";
}

function drawBackground() {
ctx.fillStyle = "#070911";
ctx.fillRect(0, 0, width, height);

for (const star of stars) {
ctx.globalAlpha = star.glow;
ctx.fillStyle = "#ffffff";
ctx.beginPath();
ctx.arc(star.x, star.y, star.r, 0, Math.PI * 2);
ctx.fill();
}
ctx.globalAlpha = 1;
}

function drawStar(cx, cy) {
const glow = ctx.createRadialGradient(cx, cy, 10, cx, cy, 110);
glow.addColorStop(0, "rgba(255, 246, 186, 1)");
glow.addColorStop(0.22, "rgba(255, 176, 55, 0.92)");
glow.addColorStop(0.55, "rgba(255, 112, 42, 0.32)");
glow.addColorStop(1, "rgba(255, 112, 42, 0)");

ctx.fillStyle = glow;
ctx.beginPath();
ctx.arc(cx, cy, 110, 0, Math.PI * 2);
ctx.fill();

ctx.fillStyle = "#ffd36b";
ctx.beginPath();
ctx.arc(cx, cy, 38, 0, Math.PI * 2);
ctx.fill();
}

function drawOrbit(cx, cy, radius) {
ctx.strokeStyle = "rgba(255, 255, 255, 0.16)";
ctx.lineWidth = 1;
ctx.beginPath();
ctx.arc(cx, cy, radius, 0, Math.PI * 2);
ctx.stroke();
}

function drawCollector(x, y, angle) {
ctx.save();
ctx.translate(x, y);
ctx.rotate(angle);
ctx.fillStyle = "#a7d8ff";
ctx.strokeStyle = "rgba(255, 255, 255, 0.65)";
ctx.lineWidth = 1;
ctx.fillRect(-5, -2, 10, 4);
ctx.strokeRect(-5, -2, 10, 4);
ctx.restore();
}

function drawEnergyBeam(cx, cy, x, y, opacity) {
ctx.strokeStyle = `rgba(93, 220, 255, ${opacity})`;
ctx.lineWidth = 1;
ctx.beginPath();
ctx.moveTo(x, y);
ctx.lineTo(cx, cy);
ctx.stroke();
}

function drawSwarm(settings) {
const cx = width / 2;
const cy = height / 2;
const radius = Math.min(settings.orbit, Math.min(width, height) * 0.42);
const count = settings.collectors;
const keplerSpeed = settings.speed * Math.sqrt(180 / settings.orbit);

drawOrbit(cx, cy, radius);
drawStar(cx, cy);

for (let i = 0; i < count; i += 1) {
const phase = (i / count) * Math.PI * 2;
const wobble = Math.sin(time * 0.8 + i) * 5;
const angle = phase + time * keplerSpeed * (0.14 + (i % 5) * 0.01);
const x = cx + Math.cos(angle) * (radius + wobble);
const y = cy + Math.sin(angle) * (radius * 0.55 + wobble);

if (i % 9 === 0) {
drawEnergyBeam(cx, cy, x, y, 0.16 + settings.efficiency / 700);
}
drawCollector(x, y, angle + Math.PI / 2);
}
}

function animate() {
const settings = readSettings();
time += 0.016;
updateLabels(settings);
drawBackground();
drawSwarm(settings);
requestAnimationFrame(animate);
}

window.addEventListener("resize", resize);
resize();
animate();
</script>
